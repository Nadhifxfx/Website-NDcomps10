// 2. CLIENT SIDE (components/VideoPagination.tsx)
function VideoPagination({ videos, page, perPage }) {
  // Calculate pagination
  const totalPages = Math.ceil(videos.length / perPage)
  const startIndex = (page - 1) * perPage
  const endIndex = startIndex + perPage
  
  // Slice array
  const paginatedVideos = videos.slice(startIndex, endIndex)
  
  // Display
  return (
    <>
      <VideoGrid videos={paginatedVideos} />
      <Pagination 
        currentPage={page}
        totalPages={totalPages}
        onPageChange={handlePageChange}
      />
    </>
  )
}

// 3. PAGINATION COMPONENT (components/Pagination.tsx)
function Pagination({ currentPage, totalPages, onPageChange }) {
  const handlePageChange = (newPage) => {
    // Update URL
    router.push(`/videos?page=${newPage}&perPage=${perPage}`)
    // Scroll to top
    window.scrollTo({ top: 0, behavior: 'smooth' })
  }
  
  return (
    <div>
      <button onClick={() => handlePageChange(currentPage - 1)}>
        Previous
      </button>
      {/* Page numbers */}
      <button onClick={() => handlePageChange(currentPage + 1)}>
        Next
      </button>
    </div>
  )
}
